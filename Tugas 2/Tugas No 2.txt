uses crt;
var
   nilaix,nilaiy : integer;
   hasiltambah,hasilkurang,hasilkali,hasildiv,hasilmod:real;
begin
     write('masukan nilaix=');readln(nilaix);
     write('masukan nilaiy=');readln(nilaiy);
     hasiltambah:=nilaix+nilaiy;
     hasilkurang:=nilaix-nilaiy;
     hasilkali:=nilaix*nilaiy;
     hasildiv:=nilaix div nilaiy;
     hasilmod:=nilaix mod nilaiy;
     writeln(nilaix,'+',nilaiy,'=',hasiltambah:2:2);
     writeln(nilaix,'-',nilaiy,'=',hasilkurang:2:2);
     writeln(nilaix,'*',nilaiy,'=',hasilkali:2:2);
     writeln(nilaix,'div',nilaiy,'=',hasildiv:2:2);
     writeln(nilaix,'mod',nilaiy,'=',hasilmod:2:2);
readkey;
end.
